import { useState, useMemo, useCallback } from 'react';
import createContextHook from '@nkzw/create-context-hook';
import { MOCK_PRODUCTS } from '@/mocks/products';
import { Product as BaseProduct, FlashSale } from '@/types';

export interface Product {
  id: string;
  name: string;
  category: string;
  province: string;
  pricePerUnit: number;
  unit: string;
  quantity: number;
  farmerName: string;
  city: string;
  createdAt: string;
  photoProduct: string;
  vendeurPhoto: string;
  vendeurNote: number;
  flashSale?: FlashSale;
}

export interface CartItem {
  product: Product;
  quantity: number;
}

function mapProduct(p: BaseProduct): Product {
  return {
    id: p.id,
    name: p.nom,
    category: p.categorie.toLowerCase(),
    province: p.province,
    pricePerUnit: p.flashSale?.active ? p.flashSale.discountedPrice : p.prix,
    unit: p.unite,
    quantity: p.quantite,
    farmerName: p.vendeurNom,
    city: p.coordonneesGPS.village,
    createdAt: p.createdAt,
    photoProduct: p.photoProduct,
    vendeurPhoto: p.vendeurPhoto,
    vendeurNote: p.vendeurNote,
    flashSale: p.flashSale,
  };
}

export const [AppProvider, useApp] = createContextHook(() => {
  const [rawProducts, setRawProducts] = useState<BaseProduct[]>(MOCK_PRODUCTS);

  const products = useMemo<Product[]>(
    () => rawProducts.filter(p => p.statut === 'disponible').map(mapProduct),
    [rawProducts]
  );

  const flashSaleProducts = useMemo<Product[]>(() => {
    const now = new Date().getTime();
    return products.filter(p => {
      if (!p.flashSale?.active) return false;
      const expires = new Date(p.flashSale.expiresAt).getTime();
      return expires > now;
    });
  }, [products]);

  const activateFlashSale = useCallback((productId: string, discountPercent: number) => {
    const now = new Date();
    const expiresAt = new Date(now.getTime() + 48 * 60 * 60 * 1000);
    setRawProducts(prev =>
      prev.map(p => {
        if (p.id !== productId) return p;
        const discountedPrice = Math.round(p.prix * (1 - discountPercent / 100));
        const flashSale: FlashSale = {
          active: true,
          discountPercent,
          originalPrice: p.prix,
          discountedPrice,
          startedAt: now.toISOString(),
          expiresAt: expiresAt.toISOString(),
        };
        return { ...p, flashSale };
      })
    );
  }, []);

  const cancelFlashSale = useCallback((productId: string) => {
    setRawProducts(prev =>
      prev.map(p => {
        if (p.id !== productId) return p;
        return { ...p, flashSale: undefined };
      })
    );
  }, []);

  const [cart, setCart] = useState<CartItem[]>([]);

  const addToCart = useCallback((product: Product, qty: number) => {
    setCart(prev => {
      const existing = prev.find(i => i.product.id === product.id);
      if (existing) {
        return prev.map(i =>
          i.product.id === product.id
            ? { ...i, quantity: i.quantity + qty }
            : i
        );
      }
      return [...prev, { product, quantity: qty }];
    });
  }, []);

  const removeFromCart = useCallback((productId: string) => {
    setCart(prev => prev.filter(i => i.product.id !== productId));
  }, []);

  const clearCart = useCallback(() => {
    setCart([]);
  }, []);

  const cartTotal = useMemo(
    () => cart.reduce((s, i) => s + i.product.pricePerUnit * i.quantity, 0),
    [cart]
  );

  return useMemo(() => ({
    products,
    flashSaleProducts,
    cart,
    addToCart,
    removeFromCart,
    clearCart,
    cartTotal,
    activateFlashSale,
    cancelFlashSale,
  }), [products, flashSaleProducts, cart, addToCart, removeFromCart, clearCart, cartTotal, activateFlashSale, cancelFlashSale]);
});

export const [AuthProvider, useAuth] = createContextHook(() => {
  const [user, setUser] = useState<User | null>(null);
  const [isLoading, setIsLoading] = useState(true);
  const queryClient = useQueryClient();

  const userQuery = useQuery({
    queryKey: ['auth_user'],
    queryFn: async () => {
      const stored = await AsyncStorage.getItem(STORAGE_KEY);
      if (stored) {
        return JSON.parse(stored) as User;
      }
      return null;
    },
  });

  useEffect(() => {
    if (!userQuery.isLoading) {
      if (userQuery.data) {
        setUser(userQuery.data);
      }
      setIsLoading(false);
    }
  }, [userQuery.data, userQuery.isLoading]);

  const loginMutation = useMutation({
    mutationFn: async ({ telephone, role }: { telephone: string; role?: UserRole }) => {
      await AsyncStorage.removeItem(STORAGE_KEY);
      await new Promise(resolve => setTimeout(resolve, 1500));
      const selectedRole = role ?? 'acheteur';
      console.log('[Auth] Creating user with role:', selectedRole);
      const newUser: User = {
        id: 'u_' + Date.now(),
        nom: '',
        prenom: '',
        ville: '',
        quartier: '',
        telephone,
        role: selectedRole,
        photoProfile: '',
        cniVerified: false,
        province: '',
        walletSolde: 0,
        note: 0,
        createdAt: new Date().toISOString(),
        kycStatus: selectedRole ? 'document_pending' : 'none',
        kycData: null,
      };
      return newUser;
    },
    onSuccess: async (data) => {
      setUser(data);
      await AsyncStorage.setItem(STORAGE_KEY, JSON.stringify(data));
      console.log('[Auth] Login success, role:', data.role);
    },
  });

  const verifyOtpMutation = useMutation({
    mutationFn: async ({ _otp }: { _otp: string }) => {
      await new Promise(resolve => setTimeout(resolve, 1000));
      return true;
    },
  });

  const setRoleMutation = useMutation({
    mutationFn: async (role: UserRole) => {
      if (!user) throw new Error('No user');
      const updated = { ...user, role };
      await AsyncStorage.setItem(STORAGE_KEY, JSON.stringify(updated));
      return updated;
    },
    onSuccess: (data) => {
      setUser(data);
      void queryClient.invalidateQueries({ queryKey: ['auth_user'] });
    },
  });

  const updateWalletMutation = useMutation({
    mutationFn: async (amount: number) => {
      if (!user) throw new Error('No user');
      const updated = { ...user, walletSolde: user.walletSolde + amount };
      await AsyncStorage.setItem(STORAGE_KEY, JSON.stringify(updated));
      return updated;
    },
    onSuccess: (data) => {
      setUser(data);
    },
  });

  const logoutMutation = useMutation({
    mutationFn: async () => {
      await AsyncStorage.removeItem(STORAGE_KEY);
    },
    onSuccess: () => {
      setUser(null);
      queryClient.clear();
    },
  });

  const completeOnboarding = useCallback(async (role: UserRole) => {
    const currentUser = user;
    if (!currentUser) {
      const stored = await AsyncStorage.getItem(STORAGE_KEY);
      if (stored) {
        const parsedUser = JSON.parse(stored) as User;
        const updated = { ...parsedUser, role, kycStatus: 'document_pending' as KycStatus };
        await AsyncStorage.setItem(STORAGE_KEY, JSON.stringify(updated));
        setUser(updated);
        return;
      }
      return;
    }
    const updated = { ...currentUser, role, kycStatus: 'document_pending' as KycStatus };
    await AsyncStorage.setItem(STORAGE_KEY, JSON.stringify(updated));
    setUser(updated);
  }, [user]);

  const updatePersonalInfo = useCallback(async (info: { nom: string; prenom: string; ville: string; quartier: string }) => {
    if (!user) return;
    const updated: User = {
      ...user,
      nom: info.nom,
      prenom: info.prenom,
      ville: info.ville,
      quartier: info.quartier,
    };
    await AsyncStorage.setItem(STORAGE_KEY, JSON.stringify(updated));
    setUser(updated);
    console.log('[Auth] Personal info updated:', info.nom, info.prenom);
  }, [user]);

  const updateProvince = useCallback(async (province: string) => {
    if (!user) return;
    const updated: User = { ...user, province };
    await AsyncStorage.setItem(STORAGE_KEY, JSON.stringify(updated));
    setUser(updated);
    console.log('[Auth] Province updated:', province);
  }, [user]);

  const updateKyc = useCallback(async (kycStatus: KycStatus, kycData: Partial<KycData>) => {
    if (!user) return;
    const mergedKyc: KycData = { ...(user.kycData ?? DEFAULT_KYC_DATA), ...kycData };
    const updated: User = {
      ...user,
      kycStatus,
      kycData: mergedKyc,
      nom: kycData.extractedName ?? user.nom,
      photoProfile: kycData.selfiePhoto ?? user.photoProfile,
      cniVerified: kycStatus === 'verified',
      province: kycData.gpsProvince ?? user.province,
    };
    await AsyncStorage.setItem(STORAGE_KEY, JSON.stringify(updated));
    setUser(updated);
  }, [user]);

  return useMemo(() => ({
    user,
    isLoading,
    isAuthenticated: !!user,
    login: loginMutation,
    verifyOtp: verifyOtpMutation,
    setRole: setRoleMutation,
    updateWallet: updateWalletMutation,
    logout: logoutMutation,
    completeOnboarding,
    updatePersonalInfo,
    updateProvince,
    updateKyc,
  }), [user, isLoading, loginMutation, verifyOtpMutation, setRoleMutation, updateWalletMutation, logoutMutation, completeOnboarding, updatePersonalInfo, updateProvince, updateKyc]);
});
